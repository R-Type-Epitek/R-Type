18 void registerComponent()
19 {
20 const char* typeName = typeid(T).name();
21
22 assert(m_componentTypes.find(typeName) == m_componentTypes.end() && "Registering component type more than once.");
23
24 m_componentTypes.insert({typeName, m_nextComponentType});
25 m_componentArrays.insert({typeName, std::make_shared<ComponentArray<T>>()});
26 ++m_nextComponentType;
27 }

29 template<typename T>
30 ComponentType getComponentType()
31 {
32 const char* typeName = typeid(T).name();
33
34 assert(m_componentTypes.find(typeName) != m_componentTypes.end() && "ComponentRType not registered before use.");
35
36 // Return this component's type - used for creating signatures
37 return m_componentTypes[typeName];
38 }

40 template<typename T>
