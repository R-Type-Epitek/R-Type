57 static std::vector<char> serializeEntity(
58 ECS::ComponentManager &componentManager,
59 GameEngine::ECS::Entity entity)
60 {
61 std::vector<char> serializedData;
62 std::ostringstream oss;
63 boost::archive::binary_oarchive archive(oss);
64
65 Serializer::serializeComponent<ComponentRType::NetworkedEntity>(componentManager, entity, archive);
66 Serializer::serializeComponent<ComponentRType::MetaData>(componentManager, entity, archive);
67
68 std::string dataStr = oss.str();
69 serializedData.assign(dataStr.begin(), dataStr.end());
70 return serializedData;
71 }
72
73 void deserializeEntity(
74 GameEngine::ECS::Registry &registry,
75 ECS::ComponentManager &componentManager,
76 const std::vector<char> &serializedData)
77 {
78 std::istringstream iss(std::string(serializedData.begin(), serializedData.end()));
79 boost::archive::binary_iarchive archive(iss);
80
81 auto componentNE = Serializer::deserializeComponent<ComponentRType::NetworkedEntity>(archive);
82 auto entity = getEntity(registry, componentManager, componentNE);
83
84 Serializer::deserializeComponentToEntity<ComponentRType::MetaData>(componentManager, entity, archive);
85 }

87 private:
88 ECS::Entity getEntity(
89 GameEngine::ECS::Registry &registry,
90 ECS::ComponentManager &componentManager,
91 ComponentRType::NetworkedEntity componentId)
92 {
93 GameEngine::ECS::Entity entity = 0;
94 auto entityOpt = getNetworkedEntityById(componentId, componentManager);
95
96 if (entityOpt.has_value()) {
97 entity = entityOpt.value();
98 } else {
99 entity = registry.createEntity();
100 registry.addComponent(entity, std::move(componentId));
101 registry.addComponent(entity, ComponentRType::MetaData {});
102 }
103 return entity;
104 }

106 std::optional<const ECS::Entity> getNetworkedEntityById(
107 ComponentRType::NetworkedEntity id,
108 ECS::ComponentManager &componentManager)
109 {
110 auto it = std::find_if(
111 m_entities.begin(),
112 m_entities.end(),
113 [&componentManager, &id](const GameEngine::ECS::Entity &entity) {
114 auto currentId = componentManager.getComponent<ComponentRType::NetworkedEntity>(entity);
115 return id == currentId;
116 });
117
118 static std::optional<const GameEngine::ECS::Entity> emptyOptional;
119 return (it != m_entities.end()) ? std::ref(*it) : emptyOptional;
120 }
