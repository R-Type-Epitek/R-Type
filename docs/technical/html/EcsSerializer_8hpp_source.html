58 void deserialize(
59 const std::vector<std::vector<char>> &serializedData,
60 Entity::EntityFactory &entityFactory)
61 {
62 std::set<ECS::Entity> activeEntities;
63
64 try {
65 for (const auto &serializedEntityData : serializedData) {
66 activeEntities.insert(deserializeEntity(serializedEntityData, entityFactory));
67 }
68 cleanUpEntity(activeEntities);
69 } catch (const boost::archive::archive_exception &e) {
70 spdlog::error("Caught boost::archive::archive_exception: {}", e.what());
71 }
72 }

75 static std::vector<char> serializeEntity(ECS::ComponentManager &componentManager, ECS::Entity entity)
76 {
77 std::vector<char> serializedData;
78 std::ostringstream oss;
79 boost::archive::binary_oarchive archive(oss);
80
81 Serializer::serializeComponent<ComponentRType::NetworkedEntity>(componentManager, entity, archive);
82 Serializer::serializeComponent<ComponentRType::MetaData>(componentManager, entity, archive);
83 Serializer::serializeComponent<ComponentRType::Position>(componentManager, entity, archive);
84 Serializer::serializeComponent<ComponentRType::Transform>(componentManager, entity, archive);
85 Serializer::serializeComponent<ComponentRType::Displayable>(componentManager, entity, archive);
86
87 std::string dataStr = oss.str();
88 serializedData.assign(dataStr.begin(), dataStr.end());
89 return serializedData;
90 }

92 ECS::Entity deserializeEntity(
93 const std::vector<char> &serializedData,
94 Entity::EntityFactory &entityFactory)
95 {
96 std::istringstream iss(std::string(serializedData.begin(), serializedData.end()));
97 boost::archive::binary_iarchive archive(iss);
98
99 auto componentNE = Serializer::deserializeComponent<ComponentRType::NetworkedEntity>(archive);
100 auto componentMD = Serializer::deserializeComponent<ComponentRType::MetaData>(archive);
101
102 auto &componentManager = getEcsRegistry().getComponentManager();
103 auto entityOpt = getNetworkedEntityById(componentNE, *componentManager);
104 if (entityOpt.has_value()) {
105 return updateEntity(entityOpt.value(), archive);
106 }
107 return createEntity(entityFactory, componentNE, componentMD, archive);
108 }

110 private:
111 ECS::Entity updateEntity(const ECS::Entity &entity, boost::archive::binary_iarchive &archive)
112 {
113 auto &componentManager = getEcsRegistry().getComponentManager();
114
115 Serializer::deserializeComponentToEntity<ComponentRType::Position>(*componentManager, entity, archive);
116 Serializer::deserializeComponentToEntity<ComponentRType::Transform>(*componentManager, entity, archive);
117 return entity;
118 }

120 ECS::Entity createEntity(
121 Entity::EntityFactory &entityFactory,
122 ComponentRType::NetworkedEntity componentNE,
123 ComponentRType::MetaData componentMD,
124 boost::archive::binary_iarchive &archive)
125 {
126 auto &componentManager = getEcsRegistry().getComponentManager();
127 const ECS::Entity entity = entityFactory.loadFromNetwork(componentNE, componentMD);
128
129 updateEntity(entity, archive);
130 Serializer::deserializeComponentToEntity<ComponentRType::Displayable>(
131 *componentManager,
132 entity,
133 archive);
134 auto &comp = componentManager->getComponent<ComponentRType::Displayable>(entity);
135 comp = ComponentRType::Displayable(
136 comp.assetPath,
137 {comp.rectLeft, comp.rectTop, comp.rectWidth, comp.rectHeight});
138 return entity;
139 }

141 void cleanUpEntity(std::set<ECS::Entity> &activeEntities)
142 {
143 std::set<ECS::Entity> m_garbageEntities {};
144
145 for (auto &entity : m_entities) {
146 if (activeEntities.find(entity) == activeEntities.end()) {
147 m_garbageEntities.insert(entity);
148 }
149 }
150 for (auto &garbageEntity : m_garbageEntities) {
151 m_entities.erase(garbageEntity);
152 getEcsRegistry().destroyEntity(garbageEntity);
153 }
154 }
155
156 std::optional<const ECS::Entity> getNetworkedEntityById(
157 ComponentRType::NetworkedEntity id,
158 ECS::ComponentManager &componentManager)
159 {
160 auto it = std::find_if(
161 m_entities.begin(),
162 m_entities.end(),
163 [&componentManager, &id](const ECS::Entity &entity) {
164 auto currentId = componentManager.getComponent<ComponentRType::NetworkedEntity>(entity);
165 return id == currentId;
166 });
167
168 static std::optional<const ECS::Entity> emptyOptional;
169 return (it != m_entities.end()) ? std::ref(*it) : emptyOptional;
170 }
