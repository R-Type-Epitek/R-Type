16 template <typename T>
17 std::shared_ptr<T> registerSystem()
18 {
19 const char* typeName = typeid(T).name();
20
21 assert(m_systems.find(typeName) == m_systems.end() && "Registering system more than once.");
22
23 auto system = std::make_shared<T>();
24 m_systems.insert({typeName, system});
25 return system;
26 }

28 template<typename T>
29 void setSignature(Signature signature)
30 {
31 const char* typeName = typeid(T).name();
32
33 assert(m_systems.find(typeName) != m_systems.end() && "System used before registered.");
34
35 m_signatures.insert({typeName, signature});
36 }
37
38 void entityDestroyed(Entity entity)
39 {
40 for (auto const& pair : m_systems)
41 {
42 auto const& system = pair.second;
43
44 system->m_entities.erase(entity);
45 }
46 }
47
48 void entitySignatureChanged(Entity entity, Signature entitySignature)
49 {
50 for (auto const& pair : m_systems)
51 {
52 auto const& type = pair.first;
53 auto const& system = pair.second;
54 auto const& systemSignature = m_signatures[type];
55
56 if ((entitySignature & systemSignature) == systemSignature) {
57 system->m_entities.insert(entity);
58 }
59 else {
60 system->m_entities.erase(entity);
61 }
62 }
63 }
