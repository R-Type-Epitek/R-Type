33 GameEngine::ECS::ComponentManager& componentManager,
34 const GameEngine::ECS::Entity& entity,
35 GameEngine::Keybinds eventKey)
36 {
37 auto& transform = componentManager.getComponent<ComponentRType::Transform>(entity);
38 const float speed = transform.speed;
39
40 switch (eventKey) {
41 case GameEngine::Keybinds::Up:
42 transform.movement = sf::Vector2f(0, -speed);
43 break;
44 case GameEngine::Keybinds::Down:
45 transform.movement = sf::Vector2f(0, speed);
46 break;
47 case GameEngine::Keybinds::Left:
48 transform.movement = sf::Vector2f(-speed, 0);
49 break;
50 case GameEngine::Keybinds::Right:
51 transform.movement = sf::Vector2f(speed, 0);
52 break;
53 case GameEngine::Keybinds::Space:
54 pressSpace(componentManager, entity);
55 transform.movement = sf::Vector2f(0, 0);
56 break;
57 default:
58 break;
59 }
